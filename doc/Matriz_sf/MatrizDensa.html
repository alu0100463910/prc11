def initialize *arg
  case arg.size
    when 1
      m=arg.pop
      @data = m 
      super(m[0].size, m.size)
    when 2
      filas, columnas = arg
      @data = Array.new(filas) {Array.new(columnas)}
      super(filas, columnas)
    else
      "error"
  end
end

def traspuesta()
    c = MatrizDensa.new(@columnas, @filas)
    c.filas.times do |i|
        c.columnas.times do |j|
            c[i][j] = self[j][i]
        end
    end
    c
end

def x(value)
